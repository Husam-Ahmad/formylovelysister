(() => {
  const MAX_CLICKS = 4;
  const RETRY_MS = 10; 

  function isVisible(el) {
    if (!el || el.disabled) return false;

    const rect = el.getBoundingClientRect();
    const style = getComputedStyle(el);

    return (
      rect.width > 0 &&
      rect.height > 0 &&
      style.display !== "none" &&
      style.visibility !== "hidden" &&
      style.pointerEvents !== "none"
    );
  }

  function isCallButton(el) {
    if (!(el instanceof HTMLElement)) return false;
    if (!el.matches("button, a")) return false;
    if (!isVisible(el)) return false;

    const text = (el.innerText || el.textContent || "")
      .trim()
      .toLowerCase();

    return text === "call";
  }

  function findTargets() {
    const targets = [];
    const elements = document.querySelectorAll("button, a");

    for (const el of elements) {
      if (isCallButton(el)) {
        targets.push(el);
        if (targets.length === MAX_CLICKS) break;
      }
    }

    return targets;
  }

  function triggerClicks(targets) {
    Promise.all(
      targets.map(el => Promise.resolve().then(() => el.click()))
    );
  }

  const timer = setInterval(() => {
    const targets = findTargets();

    if (targets.length > 0) {
      clearInterval(timer);
      triggerClicks(targets);
      console.log(`Triggered ${targets.length} click(s) and stopped.`);
    }
  }, RETRY_MS);
})();

(() => {
  const MAX_CLICKS = 4;
  const RETRY_MS = 10;

  let clickCount = 0;
  const clickedElements = new WeakSet();

  function isVisible(el) {
    if (!el || el.disabled) return false;

    const rect = el.getBoundingClientRect();
    const style = getComputedStyle(el);

    return (
      rect.width > 0 &&
      rect.height > 0 &&
      style.display !== "none" &&
      style.visibility !== "hidden" &&
      style.pointerEvents !== "none"
    );
  }

  function isCallButton(el) {
    if (!(el instanceof HTMLElement)) return false;
    if (!el.matches("button, a")) return false;
    if (!isVisible(el)) return false;

    const text = (el.innerText || el.textContent || "")
      .trim()
      .toLowerCase();

    return text === "call";
  }

  function findNextTarget() {
    const elements = document.querySelectorAll("button, a");

    for (const el of elements) {
      if (isCallButton(el) && !clickedElements.has(el)) {
        return el;
      }
    }

    return null;
  }

  const timer = setInterval(() => {
    if (clickCount >= MAX_CLICKS) {
      clearInterval(timer);
      console.log(`Clicked ${clickCount} button(s). Done.`);
      return;
    }

    const target = findNextTarget();

    if (target) {
      clickedElements.add(target);
      target.click();
      clickCount++;
      console.log(`Clicked ${clickCount}/${MAX_CLICKS}`);
    }
  }, RETRY_MS);
})();
